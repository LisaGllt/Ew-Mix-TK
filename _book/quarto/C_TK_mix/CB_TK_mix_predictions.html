We want to answer the following question : Can we predict the concentrations in EPX and IMD in earthworms exposed to different ratios of the mixture ?

10.2.1 Predicted vs. Observed

Code 
dObs_ind <- df_TK_mix |> 
  mutate(
    ID = as.numeric(ID)
    )

colnames(dObs_ind)[colnames(dObs_ind) == "Weight"] <- "Weight_obs"
colnames(dObs_ind)[colnames(dObs_ind) == "CiIMD"] <- "CiIMD_obs"
colnames(dObs_ind)[colnames(dObs_ind) == "t"] <- "Time"

col_exposure <- c(Nord_aurora[4], Nord_aurora[1], Nord_frost[2])

Sim.Res.ind <- lapply(1:20, function(j) {
  read_tsv(file.path(path_mod_mix, paste0("MonteCarlo_ind_IMD_", j, ".out"))) %>%
    mutate(sim = j)
}) %>%
  bind_rows()

df_mod_ind <- NULL

for (i in 1:Nb_experiement){
  
  ID_i <- unique(dObs_ind$ID)[i]

  Sim.Res.ind_i  <- subset(Sim.Res.ind, sim == i) |> 
    dplyr::select(where(~ !all(is.na(.x))))
  
  dObs_i <- subset(dObs_ind, ID == ID_i)
  
  Time_pred = dObs_i$Time # !!!
  
  Nom_Endpoints = c("Weight", "CiIMD", "C_exposure")
  
  Nsortie = 3
  
  MPV = IC_min = IC_max = Endpoint = index = NULL
  
  for (j in 1:length(Nom_Endpoints)){
    
    index =grep(paste0("^",Nom_Endpoints[j],"_") , colnames(Sim.Res.ind_i), fixed=FALSE)

    
    Data_sim = as.matrix( Sim.Res.ind_i[ , index] )
    
    MPV    = c(  MPV,  apply(Data_sim, 2, median, na.rm=TRUE) )
    IC_min = c(IC_min,  apply(Data_sim, 2, quantile,  0.025, na.rm=TRUE) )
    IC_max = c(IC_max,  apply(Data_sim, 2, quantile,  0.975, na.rm=TRUE) ) 
    
    Endpoint = c(Endpoint, rep(Nom_Endpoints[j], ncol(Data_sim)))
  }
  
  df_mod_ind_i <- data.frame(
    predict.endpoint = MPV,
    Time    = rep(Time_pred, Nsortie),
    low     = IC_min,
    up      = IC_max,
    Endpt   = Endpoint
    )
  
  df_mod_ind_i <- df_mod_ind_i %>%
    mutate(ID=ID_i)

  df_mod_ind <- rbind(df_mod_ind, df_mod_ind_i)
}

df_mod_Weight_ind <- subset(df_mod_ind, Endpt == "Weight")
df_mod_CiIMD_ind <- subset(df_mod_ind, Endpt == "CiIMD")

df_predobs_CiIMD_ind <- dObs_ind |>

df_mod_ind <- NULL

for (i in 1:Nb_experiement){
  
  ID_i <- unique(dObs_ind$ID)[i]

  Sim.Res.ind.full_i  <- subset(Sim.Res.ind.full, sim == i) |> 
    dplyr::select(where(~ !all(is.na(.x))))
  
  dObs_i <- subset(dObs_ind, ID == ID_i)
  
  Nom_Endpoints = c("Weight", "CiEPX", "C_exposure")
  
  Nsortie = length(Nom_Endpoints)
  
  MPV = IC_min = IC_max = Endpoint = index = NULL
  
  for (j in 1:length(Nom_Endpoints)){
    
    index =grep(paste0("^",Nom_Endpoints[j],"_") , colnames(Sim.Res.ind.full_i), fixed=FALSE)

    
    Data_sim = as.matrix( Sim.Res.ind.full_i[ , index] )
    
    MPV    = c(   MPV,  apply(Data_sim, 2, median, na.rm=TRUE))
    IC_min = c(IC_min,  apply(Data_sim, 2, quantile,  0.025, na.rm=TRUE) )
    IC_max = c(IC_max,  apply(Data_sim, 2, quantile,  0.975, na.rm=TRUE) ) 
    
    Endpoint = c(Endpoint, rep(Nom_Endpoints[j], ncol(Data_sim)))
  }
  
  df_mod_ind_i <- data.frame(
    predict.endpoint = MPV,
    Time    = rep(Time_pred, Nsortie),
    low     = IC_min,
    up      = IC_max,
    Endpt   = Endpoint
    )
  
  df_mod_ind_i <- df_mod_ind_i %>%
    mutate(ID=ID_i)

  df_mod_ind <- rbind(df_mod_ind, df_mod_ind_i)
}
